##表：user(用户表)
	```
	create table`user`(
		`id` bigint unsigned auto_increment primary key,
		`country_code` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci not null comment '国际区号',
		`phone` char(11) character set utf8mb4 collate utf8mb4_unicode_ci not null comment '用户电话',
		`email` varchar(255) character set utf8mb4 collate utf8mb4_unicode_cicomment '用户邮箱',
		`username` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci not null comment '用户名',
		`password` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci not null comment '用户密码',
		`avatar` varchar(200) character set utf8mb4 collate utf8mb4_unicode_ci comment '用户头像',
		`personal_profile` varchar(200) character set utf8mb4 collate utf8mb4_unicode_ci comment '用户简介',
		`cover_image` varchar(255) character set utf8mb4 collate utf8mb4_unicode_ci comment '用户背景封面',
		`gender` tinyint default 0 comment '性别',
		`birthday` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci comment '用户生日',
		`wechat_open_id` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci not null comment 'openId',
		`wechat_union_id` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci not null comment 'unionId',
		`wechat_no` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci not null comment '微信号',
		`country` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci comment '国',
		`province` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci comment '省',
		`city` varchar(50) character set utf8mb4 collate utf8mb4_unicode_ci comment '城市、地区',
		`register_time` int not null comment '用户注册时间',
		`register_ip` varchar(100) character set utf8mb4 collate utf8mb4_unicode_ci comment '注册ip',
		`last_login_time` int not null comment '用户最后登录时间',
		`last_login_ip` varchar(100) character set utf8mb4 collate utf8mb4_unicode_ci comment '最后注册ip',
		`is_ban` tinyint not null default 0 comment '是否禁用',
		`extra` varchar(200) character set utf8mb4 collate utf8mb4_unicode_ci comment '附加信息',
		`create_time` int not null,
		`update_time` int not null,
		`is_deleted` tinyint not null default 0
	)character set utf8mb4 collate utf8mb4_unicode_ci engine innoDB;
